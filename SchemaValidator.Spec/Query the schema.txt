SELECT 
    so.name AS TableName,
    sc.name AS ColumnName,
    st.name AS DataType,
    sc.Length AS Length,
    sc.isnullable AS IsNullable
FROM   
    SysObjects so
    INNER JOIN SysColumns sc
        ON so.id = sc.id
    INNER JOIN SysTypes st
        ON st.xtype = sc.xtype AND st.xusertype = sc.xusertype
WHERE 
    so.Type = 'U'
ORDER BY 
    so.Name
